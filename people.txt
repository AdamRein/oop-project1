public class people {
    private String Name;
    private String DOB;
    
    
    // Constructor
    public people(){
        this.Name = "";
        this.DOB = "00/00/0000";
    }
    public people(String Name, String DOB){
        this.Name = Name;
        this.DOB = DOB;
    }
    
    //seting and geting varables
    public void setName(String setName) {
        this.Name = setName;
    }
     public String getName() {
        return Name;
    }
    
     public void setDOB(String setDOB) {
        this.DOB = setDOB;
    }
     public String getDOB() {
        return DOB;
    }
    
}
